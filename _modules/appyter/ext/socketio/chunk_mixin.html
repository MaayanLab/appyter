import json
import hashlib
import asyncio
import socketio
import logging
logger = logging.getLogger(__name__)


[docs]def chunk(data, chunk_size):
  for n in range(0, len(data) // chunk_size):
    yield n, data[n:n+chunk_size]



[docs]class ChunkMixin:
  ''' Works with python-socketio methods to provide chunking for large messages
  '''
  CHUNK_SIZE = 1<<20 # 1 kb

  def __init__(self, *args, **kwargs):
    super().__init__(*args, **kwargs)
    asyncio.create_task(self.chunk_receiver())


[docs]  async def chunk_receiver(self):
    ''' Process `chunked` events on the received side
    '''
    partial_data = {}
    partial_data_lock = asyncio.Lock()
    event_dispatch_queue = asyncio.Queue()
    #
    @self.event
    async def chunked(sid, data):
      # forward to/room chunked packets
      # TODO: do we need to worry about concurrency here?
      if 'to' in data and data['to'] != sid:
        return await super().emit(data, to=data['to'])
      if 'room' in data:
        return await super().emit(data, room=data['room'])
      # otherwise process them
      async with partial_data_lock:
        if data['id'] not in partial_data:
          partial_data[data['id']] = dict(total=data['total'], chunks={}, lock=asyncio.Lock())
      #
      async with partial_data[data['id']]['lock']:
        partial_data[data['id']]['chunks'][data['chunk']] = data['data']
        if partial_data[data['id']]['total'] == len(partial_data['chunks']):
          _, chunks = zip(*sorted(partial_data[data['id']]['chunks'].items()))
          del partial_data[data['id']]
          serialized = ''.join(chunks)
          sha1 = hashlib.sha1()
          sha1.update(serialized.encode())
          data_hash = sha1.hexdigest()
          if data_hash == data['id']:
            asyncio.create_task(event_dispatch_queue.put(json.loads(serialized)))
          else:
            logger.error(f"Chunked data was corrupt")
    #
    while True:
      data = await event_dispatch_queue.get()
      self.call(data['event'], data['data'])
      event_dispatch_queue.task_done()



[docs]  async def emit(self, evt, data, **kwargs):
    ''' Use `chunked` events to send in the case of large data packets
    '''
    serialized = json.dumps(dict(event=evt, data=data), sort_keys=True, separators=None, ensure_ascii=True)
    if len(serialized) > ChunkMixin.CHUNK_SIZE:
      logger.debug(f"Large packet ({len(serialized)}), chunking...")
      sha1 = hashlib.sha1()
      sha1.update(serialized.encode())
      data_hash = sha1.hexdigest()
      n_chunks = len(serialized) // ChunkMixin.CHUNK_SIZE
      for n, chunked in chunk(serialized, ChunkMixin.CHUNK_SIZE):
        await super().emit('chunked', dict(id=data_hash, chunk=n, total=n_chunks, data=chunked, **kwargs))
    else:
      await super().emit(evt, data, **kwargs)
