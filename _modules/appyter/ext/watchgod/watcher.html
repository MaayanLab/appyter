[docs]  def should_watch_file(self, entry):
    included = any(
      fnmatch.fnmatch(entry.name, g)
      for g in self.include_file_glob
    )
    excluded = any(
      fnmatch.fnmatch(entry.name, g)
      for g in self.exclude_file_glob
    )
    return included and not excluded
